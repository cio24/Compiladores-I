/% testeo de LOOP UNTIL en donde se puede observar que funciona bien el anidamiento de los mismos%/

ULONGINT counter1;
counter1 = 0_ul;

ULONGINT counter2;
counter2 = 0_ul;

LOOP{
    OUT('loop externo');
    counter1 = counter1 + 1_ul;
    LOOP{
        OUT('    loop  interno');
        counter2 = counter2 + 1_ul;
    }
    UNTIL(counter2 == 3_ul);
    counter2 = 0_ul;
}
UNTIL(counter1 == 4_ul);
