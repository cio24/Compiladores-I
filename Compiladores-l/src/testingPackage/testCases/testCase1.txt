ULONGINT counter;
ULONGINT fibonacci;
ULONGINT target;
ULONGINT aux;
ULONGINT fibonacci_prev;

fibonacci_prev = 0_ul;
target = 10_ul;
fibonacci = 1_ul;
counter = 1_ul;

LOOP{
    aux = fibonacci;
    fibonacci = fibonacci + fibonacci_prev;
    fibonacci_prev = aux;
    counter = counter + 1_ul;
}
UNTIL(counter < target);


IF(fibonacci == 55_ul)
    OUT('fibonacci es 55 xD');
ELSE{
    IF(fibonacci == 34_ul)
        OUT('fibonacci es 34 xD');
    ELSE
        OUT('nunca entro a ningun if xD');
    END_IF;

}
END_IF;
